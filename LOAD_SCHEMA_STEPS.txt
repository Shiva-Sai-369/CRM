📍 STEP 1: Open Supabase Dashboard
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

In your browser:
  Go to: https://app.supabase.com/

Login with your credentials (you should already be logged in)


📍 STEP 2: Select Your Project
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

You should see your project in the list:
  Project: grlwnzlxvolzwdyejaji

Click on it to open


📍 STEP 3: Find SQL Editor
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

On the left sidebar, look for one of these:
  ├─ "SQL Editor" (text)
  ├─ Database icon (looks like table/squares)
  └─ "Editor" section

Click: SQL Editor


📍 STEP 4: Create New Query
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

In the SQL Editor, look for a button:
  ├─ "New Query" (top left)
  ├─ "+ New" (top left area)
  └─ Or just click in the main text area

You should see an empty text box where you can type/paste SQL


📍 STEP 5: Get the Schema File
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Open your project folder in your code editor:
  File: supabase-schema.sql
  Location: d:\WebRockets\CRM\supabase-schema.sql

Right-click → Open with → Your code editor


📍 STEP 6: Copy All the SQL Code
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

In your code editor with supabase-schema.sql open:
  
  Press: Ctrl+A (select all)
  
  Result: All code should be highlighted in blue
  
  Press: Ctrl+C (copy)
  
  Result: Code copied to clipboard


📍 STEP 7: Paste Into Supabase
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Back in Supabase SQL Editor:
  
  Click: In the main text box (where you'll type SQL)
  
  Press: Ctrl+V (paste)
  
  Result: All the SQL code appears in the text box


📍 STEP 8: Run the Query
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Look for a button near the top:
  ├─ "Run" button (blue)
  ├─ Play icon (▶)
  ├─ Or press: Ctrl+Enter
  
  Click: RUN


📍 STEP 9: Wait for Success
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

After clicking RUN:
  
  You should see:
  ├─ Loading indicator (spinning circle)
  ├─ After few seconds: "Success" message (green)
  └─ "N statements executed" (bottom right)


📍 STEP 10: Verify Tables Created
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

On the left sidebar:
  
  Look for: "Tables" section
  
  You should see these NEW tables:
  ├─ companies ✓
  ├─ profiles ✓ (THIS WAS MISSING!)
  ├─ customers ✓
  ├─ activity_logs ✓
  └─ tasks ✓
  
  If you see all 5 → SUCCESS! ✅


❌ IF YOU SEE AN ERROR
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Error Message: "Error: Syntax Error..."
  Solution: Your SQL might be corrupted
  Try: 
    1. Copy fresh from file again
    2. Paste into new query
    3. Run again

Error Message: "Error: Already Exists"
  Solution: Schema already loaded (good!)
  Next:
    1. Close query
    2. Go to: Authentication → Users
    3. Continue to create test user

Error Message: Something else...
  Solution:
    1. Screenshot the error
    2. Read it carefully (often tells you what's wrong)
    3. If stuck, ask for help with error message
